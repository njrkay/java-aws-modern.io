@Scheduled(fixedDelay=1000)
public void poll() {
  var r = sqs
    .receiveMessage(url);
  for (var m :
    r.getMessages()) {
    process(m.getBody());
    sqs.deleteMessage(url,
      m.getReceiptHandle());
  }
}
